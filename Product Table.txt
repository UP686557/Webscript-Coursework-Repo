CREATE DATABASE products;

CREATE TABLE Product(
    ID int NOT NULL AUTO_INCREMENT,
    Name varchar(255) NOT NULL,
    Description varchar(255) NOT NULL,
    Quantity int NOT NULL,
    Price float NOT NULL,
    Photo varchar(50),
    PRIMARY KEY (ID)
);
